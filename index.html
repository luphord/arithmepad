# Discounting example - defines an interest rate curve - defines a discount function - defines a npv function
// a simple continuous discount curve r = t => 0.01 + 0.002 * t; // discount factor from rates df = t => Math.exp(-r(t)*t); // helper function sum = function(values) { return _(values).reduce((x, y) => x + y, 0); }; // expecting any cashflow element of the form {t: , v: } npv = function(cashflow) { return sum(_(cashflow).map(cf => df(cf.t) * cf.v)); };
npv([{t: 0.5, v: 100}, {t: 1, v: 100}, {t: 1.5, v: 100}, {t: 2, v: 10100}])
